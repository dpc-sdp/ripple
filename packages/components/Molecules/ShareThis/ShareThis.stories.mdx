## Email example
Most networks can use the shared page's meta tags to retrieve relevant information. However, emails need to be provided (custom) subject and body.

export const EmailTemplate = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { RplShareThis },
  template: ` <rpl-share-this
              :title="title"
              :url="url"
              :emailSubject="emailSubject"
              :emailBody="emailBody" />`
});

<Canvas>
  <Story
    name="Custom email"
    args={{
      title: 'Share this',
      url: 'https://app-vic-gov-au-develop.lagoon.vicsdp.amazee.io/',
      emailSubject: 'Email subject example',
      emailBody: 'I thought you might find this page interesting.'
    }}
  >
    {EmailTemplate.bind({})}
  </Story>
</Canvas>

export const CustomNetworkTemplate = (args, { argTypes }) => ({
