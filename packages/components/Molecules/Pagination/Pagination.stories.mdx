# Pagination
Please see the [README](https://github.com/dpc-sdp/ripple/blob/master/packages/components/Molecules/Pagination/README.md) for more information about how to install and setup the component.

## Basic example

export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { RplPagination },
  methods: {
    change (newStep) {
      console.log('Going to step: ' + newStep)
    }
  },
  template: `<rpl-pagination :totalSteps="totalSteps" :initialStep="initialStep" :stepsAround="stepsAround" @change="change" />`
});

<Canvas>
  <Story
    name="Pagination steps"
    args={{
      totalSteps: 8,
      initialStep: 1,
      stepsAround: 2
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
