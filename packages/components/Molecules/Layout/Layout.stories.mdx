## Basic example

export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { RplBaseLayout },
  template: `
    <rpl-base-layout>
      <template slot="header">{{ headerSlot }}</template>
      <p>{{ contentSlot }}</p>
      <template slot="footer">{{ footerSlot }}</template>
    </rpl-base-layout>
  `
});

      footerSlot: 'Site footer'
    }}
  >

  </Story>
</Canvas>
